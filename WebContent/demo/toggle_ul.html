方案一

    $(document).ready(function () {
        //加载时隐藏子菜单
        $('li ul').hide();
        //不包含子菜单时鼠标指针和项目图标
        $('li:not(:has(ul))')
       .css({ 'cursor': 'default', 'list-style-image': 'none' });
        //包含子菜单时鼠标指针和项目图标
        $('li:has(ul)')
        .css({ 'cursor': 'pointer', 'list-style-image': 'url(http://localhost:80/mycomponent/theme/img/plus.gif)' });
        //单击含子菜单的项
        $('li:has(ul)').click(function (event) {
            if (this == event.target) {
                if ($(this).children().is(':hidden')) {
                    $(this)
                .css('list-style-image', 'url(http://localhost:80/mycomponent/theme/img/minus.gif)')
                .children().show();
                }
                else {
                    $(this)
             .css('list-style-image', 'url(http://localhost:80/mycomponent/theme/img/plus.gif)')
             .children().hide();
                }
            }
        })
    })
