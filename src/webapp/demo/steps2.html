<script type="text/javascript">
	function up(){
        switch($("#J_setpCurrent").attr("class")){            
            case "current_2":
                $("#J_setpCurrent").attr("class","current_1");
            break;
            case "current_3":
                $("#J_setpCurrent").attr("class","current_2");
            break;
            case "current_4":
                $("#J_setpCurrent").attr("class","current_3");
                break;
            case "current_5":
                $("#J_setpCurrent").attr("class","current_4");
            break;
        };
    }
    function next(){
        switch($("#J_setpCurrent").attr("class")){
            case "current_1":
                $("#J_setpCurrent").attr("class","current_2");
            break;
            case "current_2":
                $("#J_setpCurrent").attr("class","current_3");
            break;
            case "current_3":
                $("#J_setpCurrent").attr("class","current_4");
            break;
            case "current_4":
                $("#J_setpCurrent").attr("class","current_5");
            break;
        };
    }
</script>
