    <script type="text/javascript">
var d={"name":"姓名","age":"年龄"};
var des={"name":"tom","age":10};
//将英文字段名转为中文
function format2 ( d,des ) {
	var json=eval(des);
	var keys=[];
	var values=[];
	for(var obj in d){
		for(var obj2 in json){
			if(obj==obj2){
				keys.push(json[obj2]);
				values.push(d[obj]);
				break;
			}
		}
	}
	var json2=convert(keys,values);
	//debugger;
	$("#result").text(JSON.stringify(json2));
	console.log(json2)
}

function convert(keys,values){
	var arr = {};
	for(var i=0, len=keys.length; i<len; i++){
		arr[keys[i]] = values[i];
	}
	return arr;	
}
format2 ( d,des );
</script>
