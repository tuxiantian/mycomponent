<script type="text/javascript">
/*
 * 什么是闭包:
	 当内部函数 在定义它的作用域 的外部 被引用时,就创建了该内部函数的闭包 ,
	 如果内部函数引用了位于外部函数的变量,
	 当外部函数调用完毕后,这些变量在内存不会被 释放,因为闭包需要它们.
 */
function outerFun()
{
 var a=0;
 function innerFun()
 {
  a++;
  alert(a);
 }
 return innerFun;  //注意这里
}
var obj=outerFun();
obj();  //结果为1
obj();  //结果为2
</script>
