<script type="text/javascript">
function start() {
	
	setInterval(function () {
		var now=new Date();//获取当前时间
		var end=new Date(now.getFullYear()+1,0,1);//月是从0开始计数
		var time=end.getTime()-now.getTime();
		getTimeText(time);
	}, 1000);
};
function getTimeText(time) {
	//秒数
	time/=1000;
	//天数
	var days=Math.floor(time/(60*60*24));//Math.floor是向上取整
	//剩余的小时
	var hours=Math.floor(time%(24*60*60)/(60*60));
	//剩余分钟
	var m=Math.floor(time%(24*60*60)%(60*60)/60);
	//剩余秒数
	var s=Math.floor(time%(24*60*60)%(60*60)%60);
	var timeDom=document.getElementById("time");
	timeDom.innerHTML="距离下一个元旦还有"+days+"天 "+hours+"时 "+m+"分 "+s+"秒";
}
start();
</script>
